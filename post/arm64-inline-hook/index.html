#define HOOK_DEF(ret, func, ...)

HOOK_DEF(float, hook_np,int* base1,int* base2,int* base3) {
float handle = orig_hook_np(base1,base2,base3);
using namespace std;
cout<<"got orig ret value = "<<handle<<endl;
cout<<"arg1="<<base1<<endl;

cout<<"arg3="<<base3<<endl;

}

}
